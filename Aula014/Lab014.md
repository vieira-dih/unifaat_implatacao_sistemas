### 2.3 Script de Blue/Green Switch

```bash
#!/bin/bash
# blue-green-switch.sh

set -e

BLUE_TG_ARN="arn:aws:elasticloadbalancing:us-east-1:123456789012:targetgroup/ecommerce-blue/1234567890123456"
GREEN_TG_ARN="arn:aws:elasticloadbalancing:us-east-1:123456789012:targetgroup/ecommerce-green/1234567890123456"
LISTENER_ARN="arn:aws:elasticloadbalancing:us-east-1:123456789012:listener/app/ecommerce-alb/1234567890123456/1234567890123456"

# Função para verificar health do target group
check_target_health() {
    local tg_arn=$1
    local healthy_count=$(aws elbv2 describe-target-health \
        --target-group-arn $tg_arn \
        --query 'TargetHealthDescriptions[?TargetHealth.State==`healthy`] | length(@)')
    echo $healthy_count
}

# Função para switch de tráfego
switch_traffic() {
    local target_arn=$1
    local color=$2
    
    echo "Switching traffic to $color environment..."
    
    aws elbv2 modify-listener \
        --listener-arn $LISTENER_ARN \
        --default-actions Type=forward,TargetGroupArn=$target_arn
    
    echo "Traffic switched to $color successfully!"
}

# Verificar argumentos
if [ "$1" == "blue" ]; then
    TARGET_ARN=$BLUE_TG_ARN
    COLOR="blue"
elif [ "$1" == "green" ]; then
    TARGET_ARN=$GREEN_TG_ARN
    COLOR="green"
else
    echo "Usage: $0 [blue|green]"
    exit 1
fi

# Verificar saúde do target group
echo "Checking health of $COLOR environment..."
HEALTHY_COUNT=$(check_target_health $TARGET_ARN)

if [ $HEALTHY_COUNT -gt 0 ]; then
    switch_traffic $TARGET_ARN $COLOR
    echo "Blue/Green deployment completed successfully!"
else
    echo "Error: No healthy targets in $COLOR environment"
    exit 1
fi
```

### 3.2 Script de Canary Deployment

```bash
#!/bin/bash
# canary-deployment.sh

set -e

BLUE_TG_ARN="arn:aws:elasticloadbalancing:us-east-1:123456789012:targetgroup/ecommerce-blue/1234567890123456"
GREEN_TG_ARN="arn:aws:elasticloadbalancing:us-east-1:123456789012:targetgroup/ecommerce-green/1234567890123456"
RULE_ARN="arn:aws:elasticloadbalancing:us-east-1:123456789012:listener-rule/app/ecommerce-alb/1234567890123456/1234567890123456/1234567890123456"

# Função para atualizar pesos
update_weights() {
    local blue_weight=$1
    local green_weight=$2
    
    echo "Updating traffic weights: Blue=$blue_weight%, Green=$green_weight%"
    
    aws elbv2 modify-rule \
        --rule-arn $RULE_ARN \
        --actions Type=forward,ForwardConfig='{
            "TargetGroups": [
                {
                    "TargetGroupArn": "'$BLUE_TG_ARN'",
                    "Weight": '$blue_weight'
                },
                {
                    "TargetGroupArn": "'$GREEN_TG_ARN'",
                    "Weight": '$green_weight'
                }
            ]
        }'
}

# Função para verificar métricas
check_metrics() {
    echo "Checking metrics for 2 minutes..."
    sleep 120
    
    # Verificar taxa de erro (simulado)
    ERROR_RATE=$(curl -s http://your-app.com/metrics | jq '.error_rate')
    
    if (( $(echo "$ERROR_RATE > 0.05" | bc -l) )); then
        echo "Error rate too high: $ERROR_RATE"
        return 1
    fi
    
    echo "Metrics look good. Error rate: $ERROR_RATE"
    return 0
}

# Canary deployment stages
echo "Starting Canary deployment..."

# Stage 1: 10% traffic to green
update_weights 90 10
if ! check_metrics; then
    echo "Rolling back due to high error rate"
    update_weights 100 0
    exit 1
fi

# Stage 2: 50% traffic to green
update_weights 50 50
if ! check_metrics; then
    echo "Rolling back due to high error rate"
    update_weights 100 0
    exit 1
fi

# Stage 3: 100% traffic to green
update_weights 0 100
if ! check_metrics; then
    echo "Rolling back due to high error rate"
    update_weights 100 0
    exit 1
fi

echo "Canary deployment completed successfully!"
```

## 📊 Parte 4: Monitoramento com CloudWatch

### 4.1 Custom Metrics

```javascript
// metrics.js
const AWS = require('aws-sdk');
const cloudwatch = new AWS.CloudWatch();

class MetricsCollector {
    constructor(namespace = 'ECommerce/App') {
        this.namespace = namespace;
    }
    
    async putMetric(metricName, value, unit = 'Count', dimensions = []) {
        const params = {
            Namespace: this.namespace,
            MetricData: [{
                MetricName: metricName,
                Value: value,
                Unit: unit,
                Dimensions: dimensions,
                Timestamp: new Date()
            }]
        };
        
        try {
            await cloudwatch.putMetricData(params).promise();
        } catch (error) {
            console.error('Error sending metric:', error);
        }
    }
    
    async recordRequest(endpoint, responseTime, statusCode) {
        const dimensions = [
            { Name: 'Endpoint', Value: endpoint },
            { Name: 'StatusCode', Value: statusCode.toString() }
        ];
        
        await Promise.all([
            this.putMetric('RequestCount', 1, 'Count', dimensions),
            this.putMetric('ResponseTime', responseTime, 'Milliseconds', dimensions)
        ]);
    }
    
    async recordError(errorType, endpoint) {
        const dimensions = [
            { Name: 'ErrorType', Value: errorType },
            { Name: 'Endpoint', Value: endpoint }
        ];
        
        await this.putMetric('ErrorCount', 1, 'Count', dimensions);
    }
}

module.exports = MetricsCollector;
```

### 4.2 Middleware de Monitoramento

```javascript
// monitoring-middleware.js
const MetricsCollector = require('./metrics');

const metrics = new MetricsCollector();

function monitoringMiddleware(req, res, next) {
    const startTime = Date.now();
    
    // Override res.end to capture response
    const originalEnd = res.end;
    res.end = function(...args) {
        const responseTime = Date.now() - startTime;
        const statusCode = res.statusCode;
        const endpoint = req.path;
        
        // Record metrics
        metrics.recordRequest(endpoint, responseTime, statusCode);
        
        // Record errors for 4xx and 5xx
        if (statusCode >= 400) {
            const errorType = statusCode >= 500 ? 'ServerError' : 'ClientError';
            metrics.recordError(errorType, endpoint);
        }
        
        originalEnd.apply(this, args);
    };
    
    next();
}

module.exports = monitoringMiddleware;
```

## 🔄 Parte 5: Rollback Automático

### 5.1 Script de Rollback Automático

```bash
#!/bin/bash
# auto-rollback.sh

set -e

BLUE_TG_ARN="arn:aws:elasticloadbalancing:us-east-1:123456789012:targetgroup/ecommerce-blue/1234567890123456"
GREEN_TG_ARN="arn:aws:elasticloadbalancing:us-east-1:123456789012:targetgroup/ecommerce-green/1234567890123456"
LISTENER_ARN="arn:aws:elasticloadbalancing:us-east-1:123456789012:listener/app/ecommerce-alb/1234567890123456/1234567890123456"
SNS_TOPIC="arn:aws:sns:us-east-1:123456789012:deployment-alerts"

# Função para verificar métricas
check_deployment_health() {
    echo "Checking deployment health..."
    
    # Verificar alarmes CloudWatch
    ALARM_STATE=$(aws cloudwatch describe-alarms \
        --alarm-names "ECommerce-HighErrorRate" "ECommerce-HighLatency" \
        --query 'MetricAlarms[?StateValue==`ALARM`] | length(@)')
    
    if [ $ALARM_STATE -gt 0 ]; then
        echo "Alarms triggered: $ALARM_STATE"
        return 1
    fi
    
    # Verificar health checks do target group
    UNHEALTHY_TARGETS=$(aws elbv2 describe-target-health \
        --target-group-arn $GREEN_TG_ARN \
        --query 'TargetHealthDescriptions[?TargetHealth.State!=`healthy`] | length(@)')
    
    if [ $UNHEALTHY_TARGETS -gt 0 ]; then
        echo "Unhealthy targets: $UNHEALTHY_TARGETS"
        return 1
    fi
    
    echo "Deployment health check passed"
    return 0
}

# Função para rollback
perform_rollback() {
    echo "Performing automatic rollback..."
    
    # Switch traffic back to blue
    aws elbv2 modify-listener \
        --listener-arn $LISTENER_ARN \
        --default-actions Type=forward,TargetGroupArn=$BLUE_TG_ARN
    
    # Send notification
    aws sns publish \
        --topic-arn $SNS_TOPIC \
        --subject "Automatic Rollback Performed" \
        --message "Deployment failed health checks. Traffic rolled back to previous version."
    
    echo "Rollback completed successfully"
}

# Main logic
echo "Starting deployment health monitoring..."

# Wait for deployment to stabilize
sleep 300

# Check health for 10 minutes
for i in {1..10}; do
    if check_deployment_health; then
        echo "Health check $i/10 passed"
        sleep 60
    else
        echo "Health check $i/10 failed - initiating rollback"
        perform_rollback
        exit 1
    fi
done

echo "All health checks passed - deployment successful"

# Send success notification
aws sns publish \
    --topic-arn $SNS_TOPIC \
    --subject "Deployment Successful" \
    --message "New deployment passed all health checks and is stable."
```

### 5.2 Lambda Function para Rollback

```python
# rollback-lambda.py
import json
import boto3
import os

def lambda_handler(event, context):
    """
    Lambda function triggered by CloudWatch alarms
    to perform automatic rollback
    """
    
    elbv2 = boto3.client('elbv2')
    sns = boto3.client('sns')
    
    # Configuration
    BLUE_TG_ARN = os.environ['BLUE_TARGET_GROUP_ARN']
    LISTENER_ARN = os.environ['LISTENER_ARN']
    SNS_TOPIC = os.environ['SNS_TOPIC_ARN']
    
    try:
        # Parse CloudWatch alarm
        message = json.loads(event['Records'][0]['Sns']['Message'])
        alarm_name = message['AlarmName']
        new_state = message['NewStateValue']
        
        if new_state == 'ALARM':
            print(f"Alarm {alarm_name} triggered - performing rollback")
            
            # Switch traffic back to blue environment
            elbv2.modify_listener(
                ListenerArn=LISTENER_ARN,
                DefaultActions=[
                    {
                        'Type': 'forward',
                        'TargetGroupArn': BLUE_TG_ARN
                    }
                ]
            )
            
            # Send notification
            sns.publish(
                TopicArn=SNS_TOPIC,
                Subject=f'Automatic Rollback - {alarm_name}',
                Message=f'Alarm {alarm_name} triggered automatic rollback. Traffic switched to stable version.'
            )
            
            return {
                'statusCode': 200,
                'body': json.dumps('Rollback completed successfully')
            }
        
    except Exception as e:
        print(f"Error during rollback: {str(e)}")
        
        # Send error notification
        sns.publish(
            TopicArn=SNS_TOPIC,
            Subject='Rollback Failed',
            Message=f'Automatic rollback failed: {str(e)}'
        )
        
        return {
            'statusCode': 500,
            'body': json.dumps(f'Rollback failed: {str(e)}')
        }
```

### 6.2 Validação de Health Checks

```bash
#!/bin/bash
# validate-deployment.sh

ENDPOINT="http://your-alb-dns-name.com"

# Function to check endpoint
check_endpoint() {
    local url=$1
    local expected_version=$2
    
    response=$(curl -s $url)
    version=$(echo $response | jq -r '.version')
    
    if [ "$version" == "$expected_version" ]; then
        echo "✅ $url returned expected version: $version"
        return 0
    else
        echo "❌ $url returned unexpected version: $version (expected: $expected_version)"
        return 1
    fi
}

# Check health endpoint
echo "Validating health endpoint..."
health_response=$(curl -s $ENDPOINT/health)
status=$(echo $health_response | jq -r '.status')

if [ "$status" == "healthy" ]; then
    echo "✅ Health check passed"
else
    echo "❌ Health check failed: $status"
    exit 1
fi

# Check version consistency
echo "Validating version consistency..."
for i in {1..10}; do
    if ! check_endpoint $ENDPOINT "2.0.0"; then
        echo "Version validation failed on attempt $i"
        exit 1
    fi
    sleep 1
done

echo "✅ All validations passed!"
```
