Crie `src/app.js`:
```javascript
const express = require('express');
const app = express();
const PORT = process.env.PORT || 3000;

app.use(express.json());

// Health check endpoint
app.get('/health', (req, res) => {
  res.json({
    status: 'healthy',
    timestamp: new Date().toISOString(),
    version: process.env.APP_VERSION || '1.0.0',
    environment: process.env.NODE_ENV || 'development'
  });
});

// API endpoints
app.get('/api/info', (req, res) => {
  res.json({
    message: 'Lab 012 - CI/CD Pipeline',
    version: process.env.APP_VERSION || '1.0.0',
    build: process.env.BUILD_NUMBER || 'local',
    commit: process.env.COMMIT_SHA || 'unknown'
  });
});

// Calculator endpoint for testing
app.post('/api/calculate', (req, res) => {
  const { operation, a, b } = req.body;
  
  if (!operation || a === undefined || b === undefined) {
    return res.status(400).json({ error: 'Missing parameters' });
  }
  
  let result;
  switch (operation) {
    case 'add':
      result = a + b;
      break;
    case 'subtract':
      result = a - b;
      break;
    case 'multiply':
      result = a * b;
      break;
    case 'divide':
      if (b === 0) {
        return res.status(400).json({ error: 'Division by zero' });
      }
      result = a / b;
      break;
    default:
      return res.status(400).json({ error: 'Invalid operation' });
  }
  
  res.json({ result, operation, a, b });
});

if (require.main === module) {
  app.listen(PORT, () => {
    console.log(`Server running on port ${PORT}`);
  });
}

module.exports = app;
```

### 1.3 Testes Automatizados

Crie `src/tests/app.test.js`:

const request = require('supertest');
const app = require('../app');

describe('API Endpoints', () => {
  test('GET /health should return healthy status', async () => {
    const response = await request(app)
      .get('/health')
      .expect(200);
    
    expect(response.body.status).toBe('healthy');
    expect(response.body.timestamp).toBeDefined();
  });

  test('GET /api/info should return app info', async () => {
    const response = await request(app)
      .get('/api/info')
      .expect(200);
    
    expect(response.body.message).toBe('Lab 012 - CI/CD Pipeline');
    expect(response.body.version).toBeDefined();
  });

  test('POST /api/calculate should perform calculations', async () => {
    const response = await request(app)
      .post('/api/calculate')
      .send({ operation: 'add', a: 5, b: 3 })
      .expect(200);
    
    expect(response.body.result).toBe(8);
  });

  test('POST /api/calculate should handle division by zero', async () => {
    const response = await request(app)
      .post('/api/calculate')
      .send({ operation: 'divide', a: 10, b: 0 })
      .expect(400);
    
    expect(response.body.error).toBe('Division by zero');
  });

## Parte 2: Pipeline de CI (Continuous Integration) (30 min)

### 2.1 Workflow de CI

Crie `.github/workflows/ci.yml`:
```yaml
name: Continuous Integration

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  test:
    name: Run Tests
    runs-on: ubuntu-latest
    
    strategy:
      matrix:
        node-version: [16, 18, 20]
    
    steps:
      - name: Checkout code
        uses: actions/checkout@v4
      
      - name: Setup Node.js ${{ matrix.node-version }}
        uses: actions/setup-node@v4
        with:
          node-version: ${{ matrix.node-version }}
          cache: 'npm'
          cache-dependency-path: src/package-lock.json
      
      - name: Install dependencies
        run: |
          cd src
          npm ci
      
      - name: Run linting
        run: |
          cd src
          npm run lint
      
      - name: Run tests
        run: |
          cd src
          npm run test:coverage
      
      - name: Upload coverage reports
        uses: codecov/codecov-action@v3
        with:
          file: src/coverage/lcov.info
          flags: unittests
          name: codecov-umbrella

  build:
    name: Build Application
    runs-on: ubuntu-latest
    needs: test
    
    steps:
      - name: Checkout code
        uses: actions/checkout@v4
      
      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '18'
          cache: 'npm'
          cache-dependency-path: src/package-lock.json
      
      - name: Install dependencies
        run: |
          cd src
          npm ci --only=production
      
      - name: Build application
        run: |
          cd src
          npm run build
      
      - name: Create artifact
        run: |
          tar -czf app-${{ github.sha }}.tar.gz src/
      
      - name: Upload build artifact
        uses: actions/upload-artifact@v3
        with:
          name: app-build
          path: app-${{ github.sha }}.tar.gz
          retention-days: 30

  security:
    name: Security Scan
    runs-on: ubuntu-latest
    
    steps:
      - name: Checkout code
        uses: actions/checkout@v4
      
      - name: Run security audit
        run: |
          cd src
          npm audit --audit-level moderate
      
      - name: Run Snyk security scan
        uses: snyk/actions/node@master
        env:
          SNYK_TOKEN: ${{ secrets.SNYK_TOKEN }}
        with:
          args: --severity-threshold=high
```

## Parte 3: Pipeline de CD para Staging (25 min)

### 3.1 Workflow de Deploy para Staging

Crie `.github/workflows/cd-staging.yml`:
```yaml
name: Deploy to Staging

on:
  push:
    branches: [develop]
  workflow_dispatch:

jobs:
  deploy-staging:
    name: Deploy to Staging Environment
    runs-on: ubuntu-latest
    environment: staging
    
    steps:
      - name: Checkout code
        uses: actions/checkout@v4
      
      - name: Configure AWS credentials
        uses: aws-actions/configure-aws-credentials@v4
        with:
          aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
          aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws-region: ${{ secrets.AWS_REGION }}
      
      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '18'
          cache: 'npm'
          cache-dependency-path: src/package-lock.json
      
      - name: Install dependencies
        run: |
          cd src
          npm ci --only=production
      
      - name: Build for staging
        run: |
          cd src
          export NODE_ENV=staging
          export APP_VERSION=${{ github.sha }}
          npm run build
      
      - name: Create deployment package
        run: |
          cd src
          zip -r ../deployment-staging.zip . -x "tests/*" "coverage/*" "*.test.js"
      
      - name: Deploy to S3 staging bucket
        run: |
          aws s3 sync src/ s3://lab012-staging-${{ github.run_number }} --delete
          aws s3 cp deployment-staging.zip s3://lab012-artifacts/staging/
      
      - name: Update staging environment
        run: |
          # Simular deploy para ambiente de staging
          echo "Deploying to staging environment..."
          echo "Version: ${{ github.sha }}"
          echo "Build: ${{ github.run_number }}"
      
      - name: Run smoke tests
        run: |
          # Aguardar deploy
          sleep 30
          
          # Testar endpoints básicos
          curl -f http://lab012-staging-${{ github.run_number }}.s3-website-us-east-1.amazonaws.com/health || exit 1
          echo "✅ Smoke tests passed"
      
      - name: Notify deployment
        if: always()
        run: |
          if [ "${{ job.status }}" == "success" ]; then
            echo "✅ Staging deployment successful"
          else
            echo "❌ Staging deployment failed"
          fi
```

## Parte 4: Pipeline de CD para Produção (35 min)

### 4.1 Workflow de Deploy para Produção

Crie `.github/workflows/cd-production.yml`:
```yaml
name: Deploy to Production

on:
  push:
    tags:
      - 'v*'
  workflow_dispatch:
    inputs:
      version:
        description: 'Version to deploy'
        required: true
        default: 'latest'

jobs:
  deploy-production:
    name: Deploy to Production
    runs-on: ubuntu-latest
    environment: production
    
    steps:
      - name: Checkout code
        uses: actions/checkout@v4
      
      - name: Configure AWS credentials
        uses: aws-actions/configure-aws-credentials@v4
        with:
          aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
          aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws-region: ${{ secrets.AWS_REGION }}
      
      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '18'
          cache: 'npm'
          cache-dependency-path: src/package-lock.json
      
      - name: Install dependencies
        run: |
          cd src
          npm ci --only=production
      
      - name: Run full test suite
        run: |
          cd src
          npm test
      
      - name: Build for production
        run: |
          cd src
          export NODE_ENV=production
          export APP_VERSION=${{ github.ref_name }}
          export BUILD_NUMBER=${{ github.run_number }}
          export COMMIT_SHA=${{ github.sha }}
          npm run build
      
      - name: Create production package
        run: |
          cd src
          zip -r ../deployment-production.zip . -x "tests/*" "coverage/*" "*.test.js" "node_modules/.cache/*"
      
      - name: Blue-Green Deployment Setup
        run: |
          # Verificar ambiente atual
          CURRENT_ENV=$(aws s3api get-bucket-tagging --bucket lab012-production --query 'TagSet[?Key==`Environment`].Value' --output text 2>/dev/null || echo "blue")
          
          if [ "$CURRENT_ENV" = "blue" ]; then
            NEW_ENV="green"
          else
            NEW_ENV="blue"
          fi
          
          echo "CURRENT_ENV=$CURRENT_ENV" >> $GITHUB_ENV
          echo "NEW_ENV=$NEW_ENV" >> $GITHUB_ENV
          echo "Deploying to $NEW_ENV environment"
      
      - name: Deploy to new environment
        run: |
          # Criar bucket para novo ambiente se não existir
          aws s3 mb s3://lab012-production-${{ env.NEW_ENV }} 2>/dev/null || true
          
          # Configurar static website hosting
          aws s3 website s3://lab012-production-${{ env.NEW_ENV }} \
            --index-document index.html \
            --error-document error.html
          
          # Deploy da aplicação
          aws s3 sync src/ s3://lab012-production-${{ env.NEW_ENV }} --delete
          
          # Taggar bucket com ambiente
          aws s3api put-bucket-tagging \
            --bucket lab012-production-${{ env.NEW_ENV }} \
            --tagging 'TagSet=[{Key=Environment,Value=${{ env.NEW_ENV }}},{Key=Version,Value=${{ github.ref_name }}}]'
      
      - name: Health check new environment
        run: |
          # Aguardar propagação
          sleep 60
          
          # Testar novo ambiente
          NEW_URL="http://lab012-production-${{ env.NEW_ENV }}.s3-website-us-east-1.amazonaws.com"
          
          # Health check
          curl -f $NEW_URL/health || exit 1
          
          # API test
          curl -f $NEW_URL/api/info || exit 1
          
          echo "✅ Health checks passed for ${{ env.NEW_ENV }} environment"
      
      - name: Switch traffic (Blue-Green)
        run: |
          # Atualizar DNS ou load balancer para apontar para novo ambiente
          # Para este lab, vamos simular com tags
          
          aws s3api put-bucket-tagging \
            --bucket lab012-production \
            --tagging 'TagSet=[{Key=ActiveEnvironment,Value=${{ env.NEW_ENV }}},{Key=Version,Value=${{ github.ref_name }}}]'
          
          echo "✅ Traffic switched to ${{ env.NEW_ENV }} environment"
      
      - name: Cleanup old environment
        run: |
          # Aguardar estabilização
          sleep 120
          
          # Verificar se novo ambiente está estável
          NEW_URL="http://lab012-production-${{ env.NEW_ENV }}.s3-website-us-east-1.amazonaws.com"
          
          if curl -f $NEW_URL/health; then
            echo "New environment stable, cleaning up old environment"
            # aws s3 rm s3://lab012-production-${{ env.CURRENT_ENV }} --recursive
            echo "Old environment cleanup completed"
          else
            echo "New environment unstable, keeping old environment"
            exit 1
          fi
      
      - name: Create release
        uses: actions/create-release@v1
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        with:
          tag_name: ${{ github.ref }}
          release_name: Release ${{ github.ref }}
          body: |
            ## Changes in this Release
            - Deployed to production environment: ${{ env.NEW_ENV }}
            - Build number: ${{ github.run_number }}
            - Commit: ${{ github.sha }}
          draft: false
          prerelease: false
```

### 5.2 Scripts de Infraestrutura

Crie `infrastructure/scripts/create-buckets.sh`:
```bash
#!/bin/bash
set -e

ENVIRONMENT=${1:-staging}
REGION=${AWS_REGION:-us-east-1}

echo "Creating infrastructure for $ENVIRONMENT environment..."

# Criar buckets
aws s3 mb s3://lab012-$ENVIRONMENT --region $REGION
aws s3 mb s3://lab012-$ENVIRONMENT-blue --region $REGION
aws s3 mb s3://lab012-$ENVIRONMENT-green --region $REGION
aws s3 mb s3://lab012-artifacts --region $REGION

# Configurar website hosting
for bucket in lab012-$ENVIRONMENT lab012-$ENVIRONMENT-blue lab012-$ENVIRONMENT-green; do
  aws s3 website s3://$bucket \
    --index-document index.html \
    --error-document error.html
done

# Configurar políticas públicas
for bucket in lab012-$ENVIRONMENT lab012-$ENVIRONMENT-blue lab012-$ENVIRONMENT-green; do
  cat > /tmp/bucket-policy.json << EOF
{
  "Version": "2012-10-17",
  "Statement": [
    {
      "Sid": "PublicReadGetObject",
      "Effect": "Allow",
      "Principal": "*",
      "Action": "s3:GetObject",
      "Resource": "arn:aws:s3:::$bucket/*"
    }
  ]
}
EOF

  aws s3api put-public-access-block \
    --bucket $bucket \
    --public-access-block-configuration \
    "BlockPublicAcls=false,IgnorePublicAcls=false,BlockPublicPolicy=false,RestrictPublicBuckets=false"
  
  aws s3api put-bucket-policy --bucket $bucket --policy file:///tmp/bucket-policy.json
done

echo "✅ Infrastructure created for $ENVIRONMENT"
```

### 5.3 Script de Limpeza

Crie `infrastructure/scripts/cleanup.sh`:
```bash
#!/bin/bash
set -e

ENVIRONMENT=${1:-staging}

echo "Cleaning up $ENVIRONMENT environment..."

# Listar buckets para limpeza
BUCKETS=(
  "lab012-$ENVIRONMENT"
  "lab012-$ENVIRONMENT-blue" 
  "lab012-$ENVIRONMENT-green"
  "lab012-artifacts"
)

# Esvaziar e deletar buckets
for bucket in "${BUCKETS[@]}"; do
  echo "Cleaning bucket: $bucket"
  aws s3 rm s3://$bucket --recursive 2>/dev/null || true
  aws s3 rb s3://$bucket 2>/dev/null || true
done

echo "✅ Cleanup completed for $ENVIRONMENT"
```
