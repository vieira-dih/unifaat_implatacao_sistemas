## 🚀 Parte 1: Configuração Inicial

### 1.1 Criar Tabela DynamoDB

```bash
# Criar tabela para armazenar tarefas
aws dynamodb create-table \
    --table-name Tasks \
    --attribute-definitions \
        AttributeName=taskId,AttributeType=S \
    --key-schema \
        AttributeName=taskId,KeyType=HASH \
    --billing-mode PAY_PER_REQUEST \
    --region us-east-1
```

### 1.2 Criar Role IAM para Lambda

```bash
# Criar role para Lambda
aws iam create-role \
    --role-name lambda-dynamodb-role \
    --assume-role-policy-document '{
        "Version": "2012-10-17",
        "Statement": [
            {
                "Effect": "Allow",
                "Principal": {
                    "Service": "lambda.amazonaws.com"
                },
                "Action": "sts:AssumeRole"
            }
        ]
    }'

# Anexar políticas necessárias
aws iam attach-role-policy \
    --role-name lambda-dynamodb-role \
    --policy-arn arn:aws:iam::aws:policy/service-role/AWSLambdaBasicExecutionRole

aws iam attach-role-policy \
    --role-name lambda-dynamodb-role \
    --policy-arn arn:aws:iam::aws:policy/AmazonDynamoDBFullAccess
```

## 🔧 Parte 2: Desenvolvimento das Funções Lambda

### 2.1 Função: Criar Tarefa

Criar arquivo `createTask.js`:

```javascript
const AWS = require('aws-sdk');
const { v4: uuidv4 } = require('uuid');

const dynamodb = new AWS.DynamoDB.DocumentClient();

exports.handler = async (event) => {
    console.log('Event:', JSON.stringify(event, null, 2));
    
    try {
        // Parse do body da requisição
        const body = JSON.parse(event.body);
        
        // Validação básica
        if (!body.title || !body.description) {
            return {
                statusCode: 400,
                headers: {
                    'Content-Type': 'application/json',
                    'Access-Control-Allow-Origin': '*'
                },
                body: JSON.stringify({
                    error: 'Title and description are required'
                })
            };
        }
        
        // Criar nova tarefa
        const task = {
            taskId: uuidv4(),
            title: body.title,
            description: body.description,
            status: body.status || 'pending',
            createdAt: new Date().toISOString(),
            updatedAt: new Date().toISOString()
        };
        
        // Salvar no DynamoDB
        await dynamodb.put({
            TableName: 'Tasks',
            Item: task
        }).promise();
        
        return {
            statusCode: 201,
            headers: {
                'Content-Type': 'application/json',
                'Access-Control-Allow-Origin': '*'
            },
            body: JSON.stringify({
                message: 'Task created successfully',
                task: task
            })
        };
        
    } catch (error) {
        console.error('Error:', error);
        
        return {
            statusCode: 500,
            headers: {
                'Content-Type': 'application/json',
                'Access-Control-Allow-Origin': '*'
            },
            body: JSON.stringify({
                error: 'Internal server error',
                message: error.message
            })
        };
    }
};
```

### 2.2 Função: Listar Tarefas

Criar arquivo `listTasks.js`:

```javascript
const AWS = require('aws-sdk');

const dynamodb = new AWS.DynamoDB.DocumentClient();

exports.handler = async (event) => {
    console.log('Event:', JSON.stringify(event, null, 2));
    
    try {
        // Parâmetros de query
        const queryParams = event.queryStringParameters || {};
        const limit = parseInt(queryParams.limit) || 10;
        const status = queryParams.status;
        
        let scanParams = {
            TableName: 'Tasks',
            Limit: limit
        };
        
        // Filtrar por status se fornecido
        if (status) {
            scanParams.FilterExpression = '#status = :status';
            scanParams.ExpressionAttributeNames = {
                '#status': 'status'
            };
            scanParams.ExpressionAttributeValues = {
                ':status': status
            };
        }
        
        // Buscar tarefas
        const result = await dynamodb.scan(scanParams).promise();
        
        return {
            statusCode: 200,
            headers: {
                'Content-Type': 'application/json',
                'Access-Control-Allow-Origin': '*'
            },
            body: JSON.stringify({
                tasks: result.Items,
                count: result.Count,
                scannedCount: result.ScannedCount
            })
        };
        
    } catch (error) {
        console.error('Error:', error);
        
        return {
            statusCode: 500,
            headers: {
                'Content-Type': 'application/json',
                'Access-Control-Allow-Origin': '*'
            },
            body: JSON.stringify({
                error: 'Internal server error',
                message: error.message
            })
        };
    }
};
```

### 2.3 Função: Atualizar Tarefa

Criar arquivo `updateTask.js`:

```javascript
const AWS = require('aws-sdk');

const dynamodb = new AWS.DynamoDB.DocumentClient();

exports.handler = async (event) => {
    console.log('Event:', JSON.stringify(event, null, 2));
    
    try {
        // Extrair taskId da URL
        const taskId = event.pathParameters.taskId;
        const body = JSON.parse(event.body);
        
        // Construir expressão de update
        let updateExpression = 'SET updatedAt = :updatedAt';
        let expressionAttributeValues = {
            ':updatedAt': new Date().toISOString()
        };
        
        if (body.title) {
            updateExpression += ', title = :title';
            expressionAttributeValues[':title'] = body.title;
        }
        
        if (body.description) {
            updateExpression += ', description = :description';
            expressionAttributeValues[':description'] = body.description;
        }
        
        if (body.status) {
            updateExpression += ', #status = :status';
            expressionAttributeValues[':status'] = body.status;
        }
        
        // Atualizar tarefa
        const result = await dynamodb.update({
            TableName: 'Tasks',
            Key: { taskId: taskId },
            UpdateExpression: updateExpression,
            ExpressionAttributeNames: body.status ? { '#status': 'status' } : undefined,
            ExpressionAttributeValues: expressionAttributeValues,
            ReturnValues: 'ALL_NEW'
        }).promise();
        
        return {
            statusCode: 200,
            headers: {
                'Content-Type': 'application/json',
                'Access-Control-Allow-Origin': '*'
            },
            body: JSON.stringify({
                message: 'Task updated successfully',
                task: result.Attributes
            })
        };
        
    } catch (error) {
        console.error('Error:', error);
        
        return {
            statusCode: 500,
            headers: {
                'Content-Type': 'application/json',
                'Access-Control-Allow-Origin': '*'
            },
            body: JSON.stringify({
                error: 'Internal server error',
                message: error.message
            })
        };
    }
};
```

## 🌐 Parte 4: Configurar API Gateway

### 4.1 Criar API REST

```bash
# Criar API
aws apigateway create-rest-api \
    --name "TasksAPI" \
    --description "Serverless Tasks Management API"

# Obter API ID (substitua API_ID nos próximos comandos)
API_ID=$(aws apigateway get-rest-apis --query 'items[?name==`TasksAPI`].id' --output text)
ROOT_RESOURCE_ID=$(aws apigateway get-resources --rest-api-id $API_ID --query 'items[?path==`/`].id' --output text)
```

### 4.2 Criar Recursos e Métodos

```bash
# Criar recurso /tasks
TASKS_RESOURCE_ID=$(aws apigateway create-resource \
    --rest-api-id $API_ID \
    --parent-id $ROOT_RESOURCE_ID \
    --path-part tasks \
    --query 'id' --output text)

# Criar recurso /tasks/{taskId}
TASK_ID_RESOURCE_ID=$(aws apigateway create-resource \
    --rest-api-id $API_ID \
    --parent-id $TASKS_RESOURCE_ID \
    --path-part '{taskId}' \
    --query 'id' --output text)

# Método POST /tasks (criar tarefa)
aws apigateway put-method \
    --rest-api-id $API_ID \
    --resource-id $TASKS_RESOURCE_ID \
    --http-method POST \
    --authorization-type NONE

# Método GET /tasks (listar tarefas)
aws apigateway put-method \
    --rest-api-id $API_ID \
    --resource-id $TASKS_RESOURCE_ID \
    --http-method GET \
    --authorization-type NONE \
    --request-parameters method.request.querystring.status=false,method.request.querystring.limit=false

# Método PUT /tasks/{taskId} (atualizar tarefa)
aws apigateway put-method \
    --rest-api-id $API_ID \
    --resource-id $TASK_ID_RESOURCE_ID \
    --http-method PUT \
    --authorization-type NONE \
    --request-parameters method.request.path.taskId=true

# Método DELETE /tasks/{taskId} (deletar tarefa)
aws apigateway put-method \
    --rest-api-id $API_ID \
    --resource-id $TASK_ID_RESOURCE_ID \
    --http-method DELETE \
    --authorization-type NONE \
    --request-parameters method.request.path.taskId=true
```

### 4.3 Configurar Integrações

```bash
# Obter ARN das funções Lambda
ACCOUNT_ID=$(aws sts get-caller-identity --query Account --output text)
REGION="us-east-1"

# Integração POST /tasks -> createTask
aws apigateway put-integration \
    --rest-api-id $API_ID \
    --resource-id $TASKS_RESOURCE_ID \
    --http-method POST \
    --type AWS_PROXY \
    --integration-http-method POST \
    --uri "arn:aws:apigateway:$REGION:lambda:path/2015-03-31/functions/arn:aws:lambda:$REGION:$ACCOUNT_ID:function:createTask/invocations"

# Integração GET /tasks -> listTasks
aws apigateway put-integration \
    --rest-api-id $API_ID \
    --resource-id $TASKS_RESOURCE_ID \
    --http-method GET \
    --type AWS_PROXY \
    --integration-http-method POST \
    --uri "arn:aws:apigateway:$REGION:lambda:path/2015-03-31/functions/arn:aws:lambda:$REGION:$ACCOUNT_ID:function:listTasks/invocations"

# Integração PUT /tasks/{taskId} -> updateTask
aws apigateway put-integration \
    --rest-api-id $API_ID \
    --resource-id $TASK_ID_RESOURCE_ID \
    --http-method PUT \
    --type AWS_PROXY \
    --integration-http-method POST \
    --uri "arn:aws:apigateway:$REGION:lambda:path/2015-03-31/functions/arn:aws:lambda:$REGION:$ACCOUNT_ID:function:updateTask/invocations"

# Integração DELETE /tasks/{taskId} -> deleteTask
```

## 📊 Parte 6: Monitoramento

### 6.1 Configurar CloudWatch Alarms

```bash
# Alarm para erros em Lambda
aws cloudwatch put-metric-alarm \
    --alarm-name "Lambda-Errors-createTask" \
    --alarm-description "Monitor Lambda errors" \
    --metric-name Errors \
    --namespace AWS/Lambda \
    --statistic Sum \
    --period 300 \
    --threshold 1 \
    --comparison-operator GreaterThanOrEqualToThreshold \
    --dimensions Name=FunctionName,Value=createTask \
    --evaluation-periods 1

# Alarm para duração de execução
aws cloudwatch put-metric-alarm \
    --alarm-name "Lambda-Duration-createTask" \
    --alarm-description "Monitor Lambda duration" \
    --metric-name Duration \
    --namespace AWS/Lambda \
    --statistic Average \
    --period 300 \
    --threshold 5000 \
    --comparison-operator GreaterThanThreshold \
    --dimensions Name=FunctionName,Value=createTask \
    --evaluation-periods 2
```
