## Parte 1: Configuração Básica do S3 (30 min)

### 1.1 Criação de Buckets

```bash
# Definir variáveis
export BUCKET_WEBSITE="meusite-$(date +%s)"
export BUCKET_ASSETS="assets-$(date +%s)"
export AWS_REGION="us-east-1"

# Criar bucket para website
aws s3 mb s3://$BUCKET_WEBSITE --region $AWS_REGION

# Criar bucket para assets
aws s3 mb s3://$BUCKET_ASSETS --region $AWS_REGION

# Listar buckets criados
aws s3 ls
```

### 1.2 Configuração de Website Estático

```bash
# Habilitar static website hosting
aws s3 website s3://$BUCKET_WEBSITE \
    --index-document index.html \
    --error-document error.html

# Verificar configuração
aws s3api get-bucket-website --bucket $BUCKET_WEBSITE
```

### 1.3 Configuração de Permissões Públicas

```bash
# Remover block public access (cuidado em produção!)
aws s3api put-public-access-block \
    --bucket $BUCKET_WEBSITE \
    --public-access-block-configuration \
    "BlockPublicAcls=false,IgnorePublicAcls=false,BlockPublicPolicy=false,RestrictPublicBuckets=false"

# Criar política de bucket para leitura pública
cat > bucket-policy.json << EOF
{
  "Version": "2012-10-17",
  "Statement": [
    {
      "Sid": "PublicReadGetObject",
      "Effect": "Allow",
      "Principal": "*",
      "Action": "s3:GetObject",
      "Resource": "arn:aws:s3:::$BUCKET_WEBSITE/*"
    }
  ]
}
EOF

# Aplicar política
aws s3api put-bucket-policy \
    --bucket $BUCKET_WEBSITE \
    --policy file://bucket-policy.json
```

### 2.4 JavaScript para Interação com S3 (js/app.js)

```javascript
// Configuração AWS SDK
AWS.config.update({
    region: 'us-east-1',
    credentials: new AWS.CognitoIdentityCredentials({
        IdentityPoolId: 'IDENTITY_POOL_ID' // Será configurado depois
    })
});

const s3 = new AWS.S3();
const ASSETS_BUCKET = 'ASSETS_BUCKET_NAME'; // Será substituído

// Função para upload de arquivos
async function uploadFiles() {
    const fileInput = document.getElementById('fileInput');
    const statusDiv = document.getElementById('uploadStatus');
    const files = fileInput.files;
    
    if (files.length === 0) {
        showStatus('Selecione pelo menos um arquivo!', 'error');
        return;
    }
    
    statusDiv.style.display = 'block';
    showStatus('Fazendo upload...', 'info');
    
    try {
        const uploadPromises = Array.from(files).map(file => uploadFile(file));
        await Promise.all(uploadPromises);
        
        showStatus(`${files.length} arquivo(s) enviado(s) com sucesso!`, 'success');
        loadGallery(); // Recarregar galeria
        fileInput.value = ''; // Limpar input
        
    } catch (error) {
        console.error('Erro no upload:', error);
        showStatus('Erro no upload: ' + error.message, 'error');
    }
}

// Função para upload individual
function uploadFile(file) {
    const key = `uploads/${Date.now()}-${file.name}`;
    
    const params = {
        Bucket: ASSETS_BUCKET,
        Key: key,
        Body: file,
        ContentType: file.type,
        ACL: 'public-read'
    };
    
    return s3.upload(params).promise();
}

// Função para carregar galeria
async function loadGallery() {
    try {
        const params = {
            Bucket: ASSETS_BUCKET,
            Prefix: 'uploads/'
        };
        
        const data = await s3.listObjectsV2(params).promise();
        const gallery = document.getElementById('imageGallery');
        
        gallery.innerHTML = '';
        
        data.Contents.forEach(object => {
            if (object.Key.match(/\.(jpg|jpeg|png|gif)$/i)) {
                const img = document.createElement('img');
                img.src = `https://${ASSETS_BUCKET}.s3.amazonaws.com/${object.Key}`;
                img.alt = object.Key;
                img.loading = 'lazy';
                gallery.appendChild(img);
            }
        });
        
    } catch (error) {
        console.error('Erro ao carregar galeria:', error);
    }
}

// Função para mostrar status
function showStatus(message, type) {
    const statusDiv = document.getElementById('uploadStatus');
    statusDiv.textContent = message;
    statusDiv.className = type;
    statusDiv.style.display = 'block';
    
    if (type === 'success') {
        setTimeout(() => {
            statusDiv.style.display = 'none';
        }, 3000);
    }
}

// Carregar galeria ao inicializar
document.addEventListener('DOMContentLoaded', () => {
    // loadGallery(); // Descomentará após configurar Cognito
    
    // Adicionar informações do ambiente
    const info = document.createElement('div');
    info.innerHTML = `
        <p><strong>Bucket Website:</strong> ${window.location.hostname}</p>
        <p><strong>Região:</strong> us-east-1</p>
        <p><strong>CDN:</strong> ${window.location.protocol === 'https:' ? 'Ativo' : 'Inativo'}</p>
    `;
    info.style.cssText = 'background: #e3f2fd; padding: 1rem; border-radius: 5px; margin: 1rem 0;';
    document.querySelector('#home').appendChild(info);
});
```

## Parte 4: Configuração do CloudFront (30 min)

### 4.1 Criação da Distribuição CloudFront

```bash
# Criar configuração da distribuição
cat > cloudfront-config.json << EOF
{
  "CallerReference": "lab011-$(date +%s)",
  "Comment": "Lab 011 - S3 Static Website",
  "DefaultRootObject": "index.html",
  "Origins": {
    "Quantity": 1,
    "Items": [
      {
        "Id": "S3-$BUCKET_WEBSITE",
        "DomainName": "$BUCKET_WEBSITE.s3-website-$AWS_REGION.amazonaws.com",
        "CustomOriginConfig": {
          "HTTPPort": 80,
          "HTTPSPort": 443,
          "OriginProtocolPolicy": "http-only"
        }
      }
    ]
  },
  "DefaultCacheBehavior": {
    "TargetOriginId": "S3-$BUCKET_WEBSITE",
    "ViewerProtocolPolicy": "redirect-to-https",
    "TrustedSigners": {
      "Enabled": false,
      "Quantity": 0
    },
    "ForwardedValues": {
      "QueryString": false,
      "Cookies": {
        "Forward": "none"
      }
    },
    "MinTTL": 0,
    "DefaultTTL": 86400,
    "MaxTTL": 31536000
  },
  "Enabled": true,
  "PriceClass": "PriceClass_100"
}
EOF

# Criar distribuição CloudFront
aws cloudfront create-distribution \
    --distribution-config file://cloudfront-config.json \
    --query 'Distribution.{Id:Id,DomainName:DomainName,Status:Status}' \
    --output table
```

### 4.2 Aguardar Deploy da Distribuição


## Comandos de Troubleshooting

### Problemas Comuns

#### Website não carrega
```bash
# Verificar configuração de website
aws s3api get-bucket-website --bucket $BUCKET_WEBSITE

# Verificar política do bucket
aws s3api get-bucket-policy --bucket $BUCKET_WEBSITE

# Verificar block public access
aws s3api get-public-access-block --bucket $BUCKET_WEBSITE
```

#### CloudFront não funciona
```bash
# Verificar status da distribuição
aws cloudfront get-distribution --id $DISTRIBUTION_ID \
    --query 'Distribution.{Status:Status,LastModifiedTime:LastModifiedTime}'

# Verificar configuração de origem
aws cloudfront get-distribution --id $DISTRIBUTION_ID \
    --query 'Distribution.DistributionConfig.Origins'
```

#### Upload de arquivos falha
```bash
# Verificar permissões do bucket
aws s3api get-bucket-acl --bucket $BUCKET_ASSETS

# Testar upload via CLI
echo "teste" > test.txt
aws s3 cp test.txt s3://$BUCKET_ASSETS/test.txt
rm test.txt
```
